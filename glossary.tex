	name = $\mathcal{H}_\infty$,
	description = {infinity norm ($||\cdot||_\infty$), maximum energy gain of a system or peak gain over frequency}

	name = $\mathcal{H}_2$,
	description = {2-norm ($||\cdot||_2$), RMS gain of a system}

\newglossaryentry{l1}{
	name = $\ell_1$,
	description = {1-norm ($||\cdot||_1$), maximum peak-to-peak gain}

	name = $\star$-norm,

\newglossaryentry{vary}{
	name = $\sigma_y^2$,

\newglossaryentry{varu}{
	name = $\sigma_u^2$,
	description = variance of quantizer input signal $u$
}
\newglossaryentry{vard}{
	name = $\sigma_d^2$,
	description = variance of quantization noise signal $d$
}
\newglossaryentry{meany}{
	name = $\mu_y$,
	description = mean (DC component) of quantizer output signal $y$
}
\newglossaryentry{meanu}{
	name = $\mu_u$,
	description = mean (DC component) of quantizer input signal $u$
}
